コールバック関数

コールバック関数

#include<stdio.h>

typedef void (* FUNC_POINTER)(char *); //関数ポインタのtypedef
void func1(char *s){
    printf("%s\n",s);
}

void func2(char *s){
    int i;
    for(i=0;i<2;i++){
        printf("%s\n",s);
    }
}

void func(char *s,FUNC_POINTER p){
    p(s);
}

int main(){
    FUNC_POINTER p;
    p = func1;
    func("ありがと",p);
    p = func2;

    func("Thanks",p);
    return 0;
}

func1, func2がコールバック関数 funcはコールバック関数を引数にとることで、funcが呼ばれたときの処理を変えることができる。
