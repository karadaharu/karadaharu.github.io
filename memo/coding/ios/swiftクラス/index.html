swiftクラス

class Hoge {
  var name:String
  // イニシャライザ
  // 外部引数名と内部変数名を別々にすることができる
  func init(outside inside: String){
    // selfでインスタンス変数を明示的に指す
    self.name = inside
  }

  // クラス・メソッド
  class func classMethod(){
    println("classMethod")
  }

  // クラスメソッドの呼び出し
  func method(){
    self.dynamicType.classMethod()
  }
}

var a = Hoge(outside: "aaa")

イニシャライザがないとエラー。

ただし、すべてのプロパティがデフォルト値をもっている場合、イニシャライザを明示的に実装する必要あなく、暗黙的にdesignated initializerが実行される。
